function renderSalesData(data) { const container = document.getElementById('sales-data'); if (!container) return; container.innerHTML = ''; const allVenues = [...YEAR_ROUND_VENUES, ...SEASONAL_VENUES]; allVenues.forEach(venue => { const stats = data.sales[venue] || {}; if (isSalesActive(stats)) { const card = document.createElement('div'); card.className = 'card mb-3'; let tableBody = ''; Object.entries(stats).forEach(([cat, value]) => { if (typeof value === 'object') { if (Object.values(value).some(v => v > 0)) { tableBody += `${cat}`; Object.entries(value).forEach(([subCat, subValue]) => { if (subValue > 0) { tableBody += `${subCat}${subValue}`; } }); } } else if (value > 0) { const displayValue = (cat === 'Соломка' || cat === 'Кубик') ? value.toFixed(2) : value; tableBody += `${cat}${displayValue}`; } }); card.innerHTML = `

${venue}

${tableBody}
Категория Количество
`; container.appendChild(card); } }); }
